-- Criação do banco de dados
CREATE DATABASE studying;
drop table employee;
-- Utilização do banco de dados
USE studying;

-- Criação da tabela
CREATE TABLE employee (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name NVARCHAR(255),
    phone NVARCHAR(20),
    email NVARCHAR(255),
    address NVARCHAR(255),
    number NVARCHAR(10),
    neighborhood NVARCHAR(100),
    rg NVARCHAR(20),
    cpf NVARCHAR(14)
);
drop table employee;
